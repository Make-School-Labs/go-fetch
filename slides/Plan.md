How can we use selectors in a Go program?

---

```golang
package main

import (
        "fmt"
        "github.com/gocolly/colly/v2"
)

func main() {
        c := colly.NewCollector()

        c.OnHTML("a[href]", func(e *colly.HTMLElement) {
                link := e.Attr("href")
                fmt.Printf("Link found: %q -> %s\n", e.Text, link)
        })

        c.Visit("https://hackerspaces.org/")
}
```
